<script>
(function(){
  var wrapper = document.getElementById('adsrWrapper');
  var pill = document.getElementById('adsrPill');
  var icon = document.getElementById('adsrIcon');

  function setOpen(open){
    if(open){
      wrapper.classList.add('adsr-open');
      pill.setAttribute('aria-expanded','true');
      pill.nextElementSibling.setAttribute('aria-hidden','false');
      icon.textContent = '–';
    } else {
      wrapper.classList.remove('adsr-open');
      pill.setAttribute('aria-expanded','false');
      pill.nextElementSibling.setAttribute('aria-hidden','true');
      icon.textContent = '+';
    }
  }

  pill.addEventListener('click', function(){ setOpen(!wrapper.classList.contains('adsr-open')); });
  pill.addEventListener('keydown', function(e){
    if(e.key === 'Enter' || e.key === ' ') { e.preventDefault(); setOpen(!wrapper.classList.contains('adsr-open')); }
  });
})();
</script>
